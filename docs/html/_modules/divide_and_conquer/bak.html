[文档]def divide_matrix(M):
    dim_M = get_dim(M)

    r_mid = dim_M[0]//2
    c_mid = dim_M[1]//2

    M11 = []
    M12 = []
    M21 = []
    M22 = []

    i = 0
    for r in M:
        if i < r_mid:
            M11.append(r[:c_mid])
            M12.append(r[c_mid:])
        else:
            M21.append(r[:c_mid])
            M22.append(r[c_mid:])
        i += 1
    return M11, M12, M21, M22




[文档]def add_matrix(M1, M2):
    dim_A = get_dim(M1)
    M = []
    for i in range(dim_A[0]):
        M.append(
            [M1[i][j] + M2[i][j] for j in range(dim_A[1])]
        )
    return M



[文档]def sub_matrix(M1, M2):
    dim_A = get_dim(M1)
    M = []
    for i in range(dim_A[0]):
        M.append(
            [M1[i][j] - M2[i][j] for j in range(dim_A[1])]
        )
    return M



[文档]def multi_m1_1n(M1, M2):

        M.append(list(map(lambda x:x*M1[i][0], M2[0])))

[文档]def multi_1m_mn(M1, M2):
    dim_M2 = get_dim(M2)
    temp = []
    for i in range(dim_M2[1]):
        temp.append(sum(map(lambda x:x*M2[i][0], M1[0])))
    return [temp]
