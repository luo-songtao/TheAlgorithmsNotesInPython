        self.limit = limit
        self.queue = []

        

[文档]    def put(self, x):
        """添加元素
        """
        if self.is_full():
            raise QueueFullError
        self.queue.append(x)

    

[文档]    def get(self):
        """获取元素
        """
        if self.is_empty():
            raise QueueEmptyError
        return self.queue.pop(0)

    

[文档]    def is_full(self):
        return False if self.size < self.limit else False    

    

[文档]    def is_empty(self):
        return True if self.size == 0 else False

    
    @property
    def size(self):
        return len(self.queue)
