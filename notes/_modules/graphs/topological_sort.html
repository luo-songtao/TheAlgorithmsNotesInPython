[文档]def topological_sort(graph):
    """拓扑排序
    
    对图进行深度优先搜索后，顶点的探索完成顺序的逆序，就是对该graph的一种拓扑展开顺序
    """
    d, f = dfs(graph)
    return reversed(f)
